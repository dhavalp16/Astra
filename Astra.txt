Write the PPM gradient image code

The PPM P3 format is a simple text-based image format. It starts with a header (P3, then width and height, then max color value 255), followed by one RGB integer triplet per pixel.

Your renderer will loop over every pixel and compute a color gradient: red increases left-to-right and green increases top-to-bottom.





Paste the following code into your empty main.cpp file:

```cpp
#include <iostream>

int main() {
    // Image dimensions
    int image_width = 256;
    int image_height = 256;

    // PPM header: format, dimensions, max color value
    std::cout << "P3\n" << image_width << ' ' << image_height << "\n255\n";

    // Write each pixel as an RGB triplet
    for (int j = 0; j < image_height; j++) {
        for (int i = 0; i < image_width; i++) {
            // Compute color as a fraction of position
            auto r = double(i) / (image_width - 1);
            auto g = double(j) / (image_height - 1);
            auto b = 0.0;

            // Convert [0,1] floats to [0,255] integers
            int ir = int(255.999 * r);
            int ig = int(255.999 * g);
            int ib = int(255.999 * b);

            std::cout << ir << ' ' << ig << ' ' << ib << '\n';
        }
    }
}
```







The PPM header (P3, width, height, 255) tells image viewers how to interpret the data that follows.



The nested for loops iterate over every pixel. j is the row (top to bottom) and i is the column (left to right).



Each color channel is a fraction of the pixel's position, so r increases across the width and g increases down the height.



Multiplying by 255.999 and casting to int converts the 0-to-1 float range into the 0-to-255 integer range that PPM expects.

```cpp
#ifndef VEC3_H
#define VEC3_H

#include <cmath>
#include <iostream>

class vec3 {
  public:
    double e[3];

    // Default constructor: all components zero
    vec3() : e{0,0,0} {}
    // Parameterized constructor
    vec3(double e0, double e1, double e2) : e{e0, e1, e2} {}

    // Named accessors for readability
    double x() const { return e[0]; }
    double y() const { return e[1]; }
    double z() const { return e[2]; }

    // Negation: flip all components
    vec3 operator-() const { return vec3(-e[0], -e[1], -e[2]); }
    // Index access (read-only and writable)
    double operator[](int i) const { return e[i]; }
    double& operator[](int i) { return e[i]; }

    // Add another vector to this one
    vec3& operator+=(const vec3& v) {
        e[0] += v.e[0];
        e[1] += v.e[1];
        e[2] += v.e[2];
        return *this;
    }

    // Scale this vector by a scalar
    vec3& operator*=(double t) {
        e[0] *= t;
        e[1] *= t;
        e[2] *= t;
        return *this;
    }

    // Divide by reusing multiply
    vec3& operator/=(double t) {
        return *this *= 1/t;
    }

    // Euclidean length
    double length() const {
        return std::sqrt(length_squared());
    }

    // Squared length (avoids expensive sqrt when comparing)
    double length_squared() const {
        return e[0]*e[0] + e[1]*e[1] + e[2]*e[2];
    }
};

// Type aliases so code reads naturally in different contexts
using point3 = vec3;
using color = vec3;

#endif
```



The vec3 class stores three doubles in an array. The accessors x(), y(), z() make the code read like natural 3D math instead of raw array indices.

The operator overloads (+=, *=, /=) let you write v += other instead of manually adding each component. The type aliases point3 and color are just vec3 under the hood, but they make your intent clear when reading code later.

Add vector arithmetic and math functions

The class operators only modify a vector in place. To write expressions like a + b or 2.0 * v that return new vectors, you need free-function operator overloads declared outside the class.





In vec3.h, find the line #endif at the bottom of the file. Add the following operator overloads directly above it:

```cpp
// Stream output for debugging
inline std::ostream& operator<<(std::ostream& out, const vec3& v) {
    return out << v.e[0] << ' ' << v.e[1] << ' ' << v.e[2];
}

// Arithmetic: produce new vectors from two inputs
inline vec3 operator+(const vec3& u, const vec3& v) {
    return vec3(u.e[0] + v.e[0], u.e[1] + v.e[1], u.e[2] + v.e[2]);
}

inline vec3 operator-(const vec3& u, const vec3& v) {
    return vec3(u.e[0] - v.e[0], u.e[1] - v.e[1], u.e[2] - v.e[2]);
}

// Component-wise multiply (used for color blending)
inline vec3 operator*(const vec3& u, const vec3& v) {
    return vec3(u.e[0] * v.e[0], u.e[1] * v.e[1], u.e[2] * v.e[2]);
}

// Scalar multiply (both orderings so 2.0*v and v*2.0 work)
inline vec3 operator*(double t, const vec3& v) {
    return vec3(t*v.e[0], t*v.e[1], t*v.e[2]);
}

inline vec3 operator*(const vec3& v, double t) {
    return t * v;
}

// Scalar division
inline vec3 operator/(const vec3& v, double t) {
    return (1/t) * v;
}
```



C++ requires you to define both double * vec3 and vec3 * double separately. Without both, writing v * 2.0 would compile but 2.0 * v would not. The second version just calls the first, so no logic is duplicated.





Below the operator overloads you just added (still above #endif), add the three essential math functions:

```cpp
// Dot product: measures how aligned two vectors are
inline double dot(const vec3& u, const vec3& v) {
    return u.e[0] * v.e[0]
         + u.e[1] * v.e[1]
         + u.e[2] * v.e[2];
}

// Cross product: produces a vector perpendicular to both inputs
inline vec3 cross(const vec3& u, const vec3& v) {
    return vec3(u.e[1] * v.e[2] - u.e[2] * v.e[1],
                u.e[2] * v.e[0] - u.e[0] * v.e[2],
                u.e[0] * v.e[1] - u.e[1] * v.e[0]);
}

// Normalize: scale a vector to length 1
inline vec3 unit_vector(const vec3& v) {
    return v / v.length();
}
```



The dot product returns a single number measuring how much two vectors point in the same direction. You will use it for lighting calculations and intersection math.

The cross product returns a new vector that is perpendicular to both inputs. It is useful for computing surface normals. The unit_vector() function scales any vector to length 1, which is essential when you need a pure direction without magnitude.

Create the ray header file

A ray is a line that starts at a point (the origin) and extends infinitely in one direction. The parametric equation P(t) = O + tD lets you find any point along the ray by plugging in a value for t.





In Solution Explorer, right-click the RayTracer project.



Click Add, then click New Item.



Select Header File (.h), name it ray.h, and click Add.



Delete any default content in the new file and paste the following code:

```cpp
#ifndef RAY_H
#define RAY_H

#include "vec3.h"

class ray {
  public:
    // Default constructor
    ray() {}

    // Construct a ray from an origin point and a direction vector
    ray(const point3& origin, const vec3& direction) : orig(origin), dir(direction) {}

    // Accessors for origin and direction
    const point3& origin() const { return orig; }
    const vec3& direction() const { return dir; }

    // Return the point at parameter t along the ray: P(t) = O + tD
    point3 at(double t) const {
        return orig + t * dir;
    }

  private:
    point3 orig;
    vec3 dir;
};

#endif
```



The ray class stores two pieces of data: an origin point and a direction vector. The at(double t) method computes the point along the ray at parameter t using the formula orig + t * dir.

When t is 0, you get the origin. When t is 1, you get a point one direction-length away. Negative t goes behind the camera.

Set up the virtual camera in main.cpp

A virtual camera defines where the viewer sits in 3D space and how pixel coordinates map to ray directions. You need to define the image dimensions, a viewport rectangle in front of the camera, and the spacing between pixels.





Open main.cpp and delete all its contents.



Paste the following code to set up the camera and viewport:

```cpp
#include "vec3.h"
#include "color.h"
#include "ray.h"
#include <iostream>

int main() {
    // Image dimensions (16:9 aspect ratio)
    auto aspect_ratio = 16.0 / 9.0;
    int image_width = 400;
    int image_height = int(image_width / aspect_ratio);
    if (image_height < 1) image_height = 1;

    // Camera positioned at the origin, looking down negative-z
    auto focal_length = 1.0;
    auto viewport_height = 2.0;
    auto viewport_width = viewport_height * (double(image_width) / image_height);
    auto camera_center = point3(0, 0, 0);

    // Vectors spanning the viewport edges
    auto viewport_u = vec3(viewport_width, 0, 0);
    auto viewport_v = vec3(0, -viewport_height, 0);

    // Distance between pixel centers
    auto pixel_delta_u = viewport_u / image_width;
    auto pixel_delta_v = viewport_v / image_height;

    // Location of the upper-left pixel
    auto viewport_upper_left = camera_center
                             - vec3(0, 0, focal_length)
                             - viewport_u / 2
                             - viewport_v / 2;
    auto pixel00_loc = viewport_upper_left + 0.5 * (pixel_delta_u + pixel_delta_v);
}
```



The camera sits at the origin (0, 0, 0) and looks down the negative-z axis. The viewport is a rectangle floating one focal length in front of it.

The code computes pixel_delta_u and pixel_delta_v to know how far apart adjacent pixels are in 3D space. Then pixel00_loc is the center of the top-left pixel, offset by half a pixel so rays go through pixel centers rather than corners.

The negative viewport_v means pixel rows go downward (matching screen coordinates where row 0 is the top).

Implement the sky gradient and render loop

With the camera configured, you need a function that decides what color a ray sees. For now, you will blend between white and blue based on how high or low the ray points, creating a sky gradient.





In main.cpp, add the ray_color() function above main(), between the includes and int main():

```cpp
color ray_color(const ray& r) {
    // Normalize direction to get a unit vector with y in [-1, 1]
    vec3 unit_direction = unit_vector(r.direction());
    // Map y from [-1,1] to [0,1] for blending
    auto a = 0.5 * (unit_direction.y() + 1.0);
    // Linear blend: white at bottom, blue at top
    return (1.0 - a) * color(1.0, 1.0, 1.0) + a * color(0.5, 0.7, 1.0);
```

Understand the ray-sphere intersection math

A sphere is defined by a center point and a radius. A ray hits the sphere when substituting the ray equation into the sphere equation produces a solvable quadratic.



A point P is on a sphere with center C and radius r when the distance from P to C equals r. Algebraically, that means dot(P-C, P-C) = r*r.

Substitute the ray equation P(t) = O + t*D into the sphere equation and you get a quadratic in t. Compute oc = center - origin, then: a = dot(d,d), h = dot(d,oc), c = dot(oc,oc) - r*r.

The discriminant h*h - a*c tells you if the ray hits. If it is negative, the ray misses entirely. If zero or positive, the ray intersects the sphere at one or two points.

Write the hit_sphere() function

This function takes a sphere center, radius, and ray, then returns the parameter t of the closest intersection. A return value of -1 means no hit.





In main.cpp, find the existing ray_color() function.



Add the following hit_sphere() function directly above ray_color():

```cpp
double hit_sphere(const point3& center, double radius, const ray& r) {
    vec3 oc = center - r.origin();
    auto a = dot(r.direction(), r.direction());
    auto h = dot(r.direction(), oc);
    auto c = dot(oc, oc) - radius * radius;
    auto discriminant = h * h - a * c;
    if (discriminant < 0)
        return -1.0;
    return (h - std::sqrt(discriminant)) / a;
}
```







vec3 oc = center - r.origin() computes the vector from the ray origin to the sphere center.



a, h, and c are the coefficients of the quadratic equation formed by substituting the ray into the sphere equation.



If the discriminant is negative, no real solutions exist and the ray misses the sphere.



(h - sqrtd) / a returns the nearest intersection point (the smaller t value, which is the front of the sphere).

💡 Why h instead of b?

Traditional quadratic notation uses b, but the ray tracing community simplifies with h (half-b). This avoids extra multiplications and divisions by 2, making the formula cleaner without changing the result.

Create hittable.h with the abstract interface

Every object in your scene needs to answer one question: did a given ray hit me, and if so, where? The hittable interface defines this contract as a pure virtual function that any shape must implement.





In the Solution Explorer, right-click your project name.

Replace any default content with the following code:

```cpp
#ifndef HITTABLE_H
#define HITTABLE_H

#include "ray.h"

class hit_record {
  public:
    point3 p;
    vec3 normal;
    double t;
    bool front_face;

    void set_face_normal(const ray& r, const vec3& outward_normal) {
        front_face = dot(r.direction(), outward_normal) < 0;
        normal = front_face ? outward_normal : -outward_normal;
    }
};

class hittable {
  public:
    virtual ~hittable() = default;
    virtual bool hit(const ray& r, double ray_tmin, double ray_tmax, hit_record& rec) const = 0;
};

#endif
```



The hit_record struct stores everything about a ray-object intersection: the hit point p, the surface normal, the ray parameter t, and whether the ray hit the front or back face.

The set_face_normal() method ensures normals always point against the incoming ray. This matters later for materials that behave differently on inside vs outside surfaces.

The hittable class is abstract. The = 0 makes hit() a pure virtual function. Any shape (sphere, plane, triangle) must provide its own implementation.

💡 Why does hit() take ray_tmin and ray_tmax?

These parameters restrict which intersections count. A hit at t = 0.0001 is valid but t = -3.0 is behind the camera. Using a small positive ray_tmin also prevents a ray from hitting the same surface it just bounced off of (shadow acne).

```cpp
#ifndef SPHERE_H
#define SPHERE_H

#include "hittable.h"

class sphere : public hittable {
  public:
    sphere(const point3& center, double radius) : center(center), radius(radius) {}

    bool hit(const ray& r, double ray_tmin, double ray_tmax, hit_record& rec) const override {
        vec3 oc = center - r.origin();
        auto a = dot(r.direction(), r.direction());
        auto h = dot(r.direction(), oc);
        auto c = dot(oc, oc) - radius * radius;

        auto discriminant = h * h - a * c;
        if (discriminant < 0)
            return false;

        auto sqrtd = std::sqrt(discriminant);
        auto root = (h - sqrtd) / a;
        if (root <= ray_tmin || ray_tmax <= root) {
            root = (h + sqrtd) / a;
            if (root <= ray_tmin || ray_tmax <= root)
                return false;
        }

        rec.t = root;
        rec.p = r.at(rec.t);
        vec3 outward_normal = (rec.p - center) / radius;
        rec.set_face_normal(r, outward_normal);

        return true;
    }

  private:
    point3 center;
    double radius;
};

#endif
```



This is the same quadratic formula math from your hit_sphere() function, but now packaged inside a class. The sphere stores its own center and radius as private members.

The key improvement is the bounds checking with ray_tmin and ray_tmax. If the closer root falls outside the valid range, it tries the farther root. Only valid hits get recorded.

```cpp
#ifndef HITTABLE_LIST_H
#define HITTABLE_LIST_H

#include "hittable.h"
#include <memory>
#include <vector>

using std::make_shared;
using std::shared_ptr;

class hittable_list : public hittable {
  public:
    std::vector<shared_ptr<hittable>> objects;

    hittable_list() {}
    hittable_list(shared_ptr<hittable> object) { add(object); }

    void clear() { objects.clear(); }

    void add(shared_ptr<hittable> object) {
        objects.push_back(object);
    }

    bool hit(const ray& r, double ray_tmin, double ray_tmax, hit_record& rec) const override {
        hit_record temp_rec;
        bool hit_anything = false;
        auto closest_so_far = ray_tmax;

        for (const auto& object : objects) {
            if (object->hit(r, ray_tmin, closest_so_far, temp_rec)) {
                hit_anything = true;
                closest_so_far = temp_rec.t;
                rec = temp_rec;
            }
        }

        return hit_anything;
    }
};

#endif
```



The hittable_list itself is also a hittable. Its hit() method loops through every object and tracks the closest intersection found so far.

The trick is passing closest_so_far as the upper bound for each subsequent test. Once you find a hit at t = 3.0, anything farther than 3.0 is automatically rejected. This ensures you always render the nearest object.

💡 What is shared_ptr?

A shared_ptr is a C++ smart pointer that automatically manages memory. Multiple parts of your code can share ownership of the same object, and it gets deleted only when the last reference goes away. This prevents memory leaks without manual delete calls.

A Lambertian surface scatters light equally in all directions from the hit point. The scattered ray's direction is the surface normal plus a random unit vector. This produces the characteristic soft, matte appearance you see on objects like chalk, unpolished wood, or matte paint.

Each bounce absorbs 50% of the light energy (the 0.5 multiplier). After several bounces, the remaining energy approaches zero, which is why shadows appear darker.

Add random number generation to vec3.h

Diffuse shading requires picking random directions for bounced rays. You need a random number generator and a function that produces uniformly distributed random points on a unit sphere.

After the unit_vector() function (before the #endif), add the following random utility functions:

```cpp
inline double random_double() {
    static std::uniform_real_distribution<double> distribution(0.0, 1.0);
    static std::mt19937 generator;
    return distribution(generator);
}

inline double random_double(double min, double max) {
    return min + (max - min) * random_double();
}

inline vec3 random_vec3() {
    return vec3(random_double(), random_double(), random_double());
}

inline vec3 random_vec3(double min, double max) {
    return vec3(random_double(min, max), random_double(min, max), random_double(min, max));
}

inline vec3 random_unit_vector() {
    while (true) {
        auto p = random_vec3(-1, 1);
        auto lensq = p.length_squared();
        if (1e-160 < lensq && lensq <= 1)
            return p / std::sqrt(lensq);
    }
}
```



random_double() returns a random number between 0 and 1 using C++'s std::mt19937 engine and std::uniform_real_distribution.

random_unit_vector() uses rejection sampling: it generates random points inside a cube, discards any that fall outside the unit sphere, then normalizes the result. The 1e-160 lower bound prevents division by near-zero vectors that would produce infinity.

Update color.h for multisampling and gamma correction

When you cast multiple rays per pixel and accumulate their colors, you need to divide by the sample count before outputting. You also need gamma correction because human eyes perceive brightness non-linearly.

```cpp
#ifndef COLOR_H
#define COLOR_H

#include "vec3.h"
#include <iostream>

// Clamp a value to the range [min, max]
inline double clamp(double x, double min, double max) {
    if (x < min) return min;
    if (x > max) return max;
    return x;
}

void write_color(std::ostream& out, const color& pixel_color, int samples_per_pixel) {
    auto r = pixel_color.x();
    auto g = pixel_color.y();
    auto b = pixel_color.z();

    // Divide accumulated color by number of samples
    auto scale = 1.0 / samples_per_pixel;
    r *= scale;
    g *= scale;
    b *= scale;

    // Apply gamma 2.0 correction (sqrt)
    r = std::sqrt(r);
    g = std::sqrt(g);
    b = std::sqrt(b);

    // Convert to [0, 255] byte range with clamping
    int rbyte = int(256 * clamp(r, 0.0, 0.999));
    int gbyte = int(256 * clamp(g, 0.0, 0.999));
    int bbyte = int(256 * clamp(b, 0.0, 0.999));

    out << rbyte << ' ' << gbyte << ' ' << bbyte << '\n';
}

#endif
```



Without gamma correction, your image appears too dark. Monitors apply a gamma curve when displaying pixels, so you need to "pre-correct" by raising each color channel to the power of 1/gamma. For gamma 2.0, that is simply std::sqrt().

The clamp() helper prevents out-of-range values from wrapping around and producing visual artifacts.

Make ray_color() recursive and add antialiasing

The core of diffuse shading is recursive ray bouncing. When a ray hits a surface, it bounces in a random direction (the normal plus a random unit vector). Each bounce absorbs half the light. Antialiasing fires multiple rays per pixel at slightly different sub-pixel positions and averages the results.





Replace the entire contents of main.cpp with the following code:

```cpp
#include "vec3.h"
#include "color.h"
#include "ray.h"
#include "hittable.h"
#include "hittable_list.h"
#include "sphere.h"

#include <iostream>
#include <limits>

const double infinity = std::numeric_limits<double>::infinity();

// Recursive ray_color with depth limit for diffuse bouncing
color ray_color(const ray& r, int depth, const hittable& world) {
    // Base case: exceeded bounce limit, no more light gathered
    if (depth <= 0)
        return color(0, 0, 0);

    hit_record rec;
    if (world.hit(r, 0.001, infinity, rec)) {
        // Lambertian scatter: bounce in normal + random direction
        vec3 direction = rec.normal + random_unit_vector();
        // Absorb 50% of light on each bounce
        return 0.5 * ray_color(ray(rec.p, direction), depth - 1, world);
    }

    // Sky gradient background
    vec3 unit_direction = unit_vector(r.direction());
    auto a = 0.5 * (unit_direction.y() + 1.0);
    return (1.0 - a) * color(1.0, 1.0, 1.0) + a * color(0.5, 0.7, 1.0);
}

int main() {
    // Image
    auto aspect_ratio = 16.0 / 9.0;
    int image_width = 400;
    int image_height = int(image_width / aspect_ratio);
    if (image_height < 1) image_height = 1;

    // Antialiasing and recursion settings
    int samples_per_pixel = 50;
    int max_depth = 10;

    // World
    hittable_list world;
    world.add(make_shared<sphere>(point3(0, 0, -1), 0.5));
    world.add(make_shared<sphere>(point3(0, -100.5, -1), 100));

    // Camera
    auto focal_length = 1.0;
    auto viewport_height = 2.0;
    auto viewport_width = viewport_height * (double(image_width) / image_height);
    auto camera_center = point3(0, 0, 0);

    auto viewport_u = vec3(viewport_width, 0, 0);
    auto viewport_v = vec3(0, -viewport_height, 0);

    auto pixel_delta_u = viewport_u / image_width;
    auto pixel_delta_v = viewport_v / image_height;

    auto viewport_upper_left = camera_center
                             - vec3(0, 0, focal_length)
                             - viewport_u / 2
                             - viewport_v / 2;
    auto pixel00_loc = viewport_upper_left + 0.5 * (pixel_delta_u + pixel_delta_v);

    // Render
    std::cout << "P3\n" << image_width << ' ' << image_height << "\n255\n";

    for (int j = 0; j < image_height; j++) {
        std::clog << "\rScanlines remaining: " << (image_height - j) << ' ' << std::flush;
        for (int i = 0; i < image_width; i++) {
            // Accumulate color from multiple random samples per pixel
            color pixel_color(0, 0, 0);
            for (int sample = 0; sample < samples_per_pixel; sample++) {
                // Random sub-pixel offset for antialiasing
                auto offset_x = random_double() - 0.5;
                auto offset_y = random_double() - 0.5;
                auto pixel_sample = pixel00_loc
                                  + ((i + offset_x) * pixel_delta_u)
                                  + ((j + offset_y) * pixel_delta_v);
                auto ray_direction = pixel_sample - camera_center;
                ray r(camera_center, ray_direction);
                pixel_color += ray_color(r, max_depth, world);
            }
            write_color(std::cout, pixel_color, samples_per_pixel);
        }
    }

    std::clog << "\rDone.                 \n";
}
```

ray_color() now takes an int depth parameter. When depth hits 0, it returns black (no more light). On a hit, it bounces the ray in a random Lambertian direction and recurses.

```cpp
#ifndef VEC3_H
#define VEC3_H

#include <cmath>
#include <iostream>
#include <random>

class vec3 {
  public:
    double e[3];

    vec3() : e{0,0,0} {}
    vec3(double e0, double e1, double e2) : e{e0, e1, e2} {}

    double x() const { return e[0]; }
    double y() const { return e[1]; }
    double z() const { return e[2]; }

    vec3 operator-() const { return vec3(-e[0], -e[1], -e[2]); }
    double operator[](int i) const { return e[i]; }
    double& operator[](int i) { return e[i]; }

    vec3& operator+=(const vec3& v) {
        e[0] += v.e[0];
        e[1] += v.e[1];
        e[2] += v.e[2];
        return *this;
    }

    vec3& operator*=(double t) {
        e[0] *= t;
        e[1] *= t;
        e[2] *= t;
        return *this;
    }

    vec3& operator/=(double t) {
        return *this *= 1/t;
    }

    double length() const {
        return std::sqrt(length_squared());
    }

    double length_squared() const {
        return e[0]*e[0] + e[1]*e[1] + e[2]*e[2];
    }
};

using point3 = vec3;
using color = vec3;

inline std::ostream& operator<<(std::ostream& out, const vec3& v) {
    return out << v.e[0] << ' ' << v.e[1] << ' ' << v.e[2];
}

inline vec3 operator+(const vec3& u, const vec3& v) {
    return vec3(u.e[0] + v.e[0], u.e[1] + v.e[1], u.e[2] + v.e[2]);
}

inline vec3 operator-(const vec3& u, const vec3& v) {
    return vec3(u.e[0] - v.e[0], u.e[1] - v.e[1], u.e[2] - v.e[2]);
}

inline vec3 operator*(const vec3& u, const vec3& v) {
    return vec3(u.e[0] * v.e[0], u.e[1] * v.e[1], u.e[2] * v.e[2]);
}

inline vec3 operator*(double t, const vec3& v) {
    return vec3(t*v.e[0], t*v.e[1], t*v.e[2]);
}

inline vec3 operator*(const vec3& v, double t) {
    return t * v;
}

inline vec3 operator/(const vec3& v, double t) {
    return (1/t) * v;
}

inline double dot(const vec3& u, const vec3& v) {
    return u.e[0] * v.e[0]
         + u.e[1] * v.e[1]
         + u.e[2] * v.e[2];
}

inline vec3 cross(const vec3& u, const vec3& v) {
    return vec3(u.e[1] * v.e[2] - u.e[2] * v.e[1],
                u.e[2] * v.e[0] - u.e[0] * v.e[2],
                u.e[0] * v.e[1] - u.e[1] * v.e[0]);
}

inline vec3 unit_vector(const vec3& v) {
    return v / v.length();
}
```
